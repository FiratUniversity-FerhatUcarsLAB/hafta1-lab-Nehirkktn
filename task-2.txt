import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

public class BasitKartOyunu {

    public static void main(String[] args) {

        // 1. Hazırlık Aşaması
        List<Integer> oyuncuEli = new ArrayList<>();
        List<Integer> bilgisayarEli = new ArrayList<>();
        int oyuncuPuanı = 0;
        int bilgisayarPuanı = 0;

        // Deste oluştur
        List<Integer> deste = new ArrayList<>();
        for (int i = 1; i <= 10; i++) {
            deste.add(i);
        }

        // Desteyi karıştır
        Collections.shuffle(deste);

        // Kartları dağıt
        for (int i = 0; i < 5; i++) {
            oyuncuEli.add(deste.get(i));
        }
        for (int i = 5; i < 10; i++) {
            bilgisayarEli.add(deste.get(i));
        }

        // Oyuncunun kartlarını sıralayarak görmesini kolaylaştır
        Collections.sort(oyuncuEli);

        Scanner scanner = new Scanner(System.in);

        // 2. Ana Oyun Döngüsü
        for (int tur = 1; tur <= 5; tur++) {
            System.out.println("\n----- TUR " + tur + " -----");
            System.out.println("Puanlar -> Oyuncu: " + oyuncuPuanı + ", Bilgisayar: " + bilgisayarPuanı);
            System.out.println("Elinizdeki Kartlar: " + oyuncuEli);

            // 2.2. Oyuncu Hamlesi
            int oyuncununKarti = 0;
            boolean gecerliGiris = false;
            while (!gecerliGiris) {
                System.out.print("Oynamak istediğiniz kartı seçin: ");
                try {
                    oyuncununKarti = Integer.parseInt(scanner.nextLine());
                    if (oyuncuEli.contains(oyuncununKarti)) {
                        gecerliGiris = true;
                    } else {
                        System.out.println("Hata: Elinizde bu kart yok. Lütfen listeden bir kart seçin.");
                    }
                } catch (NumberFormatException e) {
                    System.out.println("Hata: Lütfen geçerli bir sayı girin.");
                }
            }
            oyuncuEli.remove(Integer.valueOf(oyuncununKarti));

            // 2.3. Bilgisayar Stratejisi
            int oynanacakBilgisayarKarti = -1;

            // Strateji Adım 1: Kazanma ihtimalini kontrol et
            int kazanmaKarti = Integer.MAX_VALUE; // Başlangıç için mümkün olan en büyük değer
            boolean kazanmaKartiBulundu = false;

            for (Integer kart : bilgisayarEli) {
                if (kart > oyuncununKarti && kart < kazanmaKarti) {
                    kazanmaKarti = kart;
                    kazanmaKartiBulundu = true;
                }
            }

            if (kazanmaKartiBulundu) {
                // Yenebilecek en küçük kart bulundu.
                oynanacakBilgisayarKarti = kazanmaKarti;
            } else {
                // Strateji Adım 2: Kazanamıyorsan en küçük kartı feda et
                int fedaEdilecekKart = Integer.MAX_VALUE;
                for (Integer kart : bilgisayarEli) {
                    if (kart < fedaEdilecekKart) {
                        fedaEdilecekKart = kart;
                    }
                }
                oynanacakBilgisayarKarti = fedaEdilecekKart;
            }

            bilgisayarEli.remove(Integer.valueOf(oynanacakBilgisayarKarti));

            System.out.println("Siz Oynadınız: " + oyuncununKarti);
            System.out.println("Bilgisayar Oynadı: " + oynanacakBilgisayarKarti);

            // 2.4. Tur Sonucunu Değerlendir
            if (oyuncununKarti > oynanacakBilgisayarKarti) {
                oyuncuPuanı++;
                System.out.println("Bu turu KAZANDINIZ!");
            } else if (oynanacakBilgisayarKarti > oyuncununKarti) {
                bilgisayarPuanı++;
                System.out.println("Bu turu KAYBETTİNİZ.");
            } else {
                System.out.println("Bu tur BERABERE.");
            }
        }

        // 3. Oyun Sonu
        System.out.println("\n========== OYUN BİTTİ ==========");
        System.out.println("Final Skoru -> Oyuncu: " + oyuncuPuanı + ", Bilgisayar: " + bilgisayarPuanı);

        if (oyuncuPuanı > bilgisayarPuanı) {
            System.out.println("Tebrikler, oyunu KAZANDINIZ!");
        } else if (bilgisayarPuanı > oyuncuPuanı) {
            System.out.println("Bilgisayar kazandı.");
        } else {
            System.out.println("Oyun BERABERE bitti!");
        }

        scanner.close();
    }
}
